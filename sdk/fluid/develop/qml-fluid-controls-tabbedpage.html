Detailed Description


  import QtQuick 2.10
  import Fluid.Controls 1.0 as FluidControls

  FluidControls.ApplicationWindow {
      title: "Application Name"
      width: 1024
      height: 800
      visible: true

      initialPage: FluidControls.TabbedPage {
          FluidControls.Tab {
              title: "Tab 1"

              Label {
                  anchors.centerIn: parent
                  text: "Hello World!"
              }
          }

          FluidControls.Tab {
              title: "Tab 2"

              Label {
                  anchors.centerIn: parent
                  text: "Hello World!"
              }
          }
      }
  }
